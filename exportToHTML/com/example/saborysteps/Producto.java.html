package com.example.saborysteps;

public class Producto {
    private String nombre;
    private String tamaño;
    private double precio;
    private int imagenResourceId;

    public Producto(String nombre, String tamaño, double precio,int imagenResourceId) {
        this.nombre = nombre;
        this.tamaño = tamaño;
        this.precio = precio;
        this.imagenResourceId = imagenResourceId;
    }

    public String getNombre() {
        return nombre;
    }

    public String getTamaño() {
        return tamaño;
    }

    public double getPrecio() {
        return precio;
    }

    public int getImagenResourceId() {
        return imagenResourceId;
    }


}
